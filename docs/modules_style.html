  20def load_sparse(path: str, name: str):
  21    """
  22    Load a sparse matrix dataset along with associated gene
  23    and cell metadata, and return it as a dense DataFrame.
  24
  25    This function expects the input directory to contain three files in standard
  26    10x Genomics format:
  27      - "matrix.mtx": the gene expression matrix in Matrix Market format
  28      - "genes.tsv": tab-separated file containing gene identifiers
  29      - "barcodes.tsv": tab-separated file containing cell barcodes / names
  30
  31    Parameters
  32    ----------
  33    path : str
  34        Path to the directory containing the matrix and annotation files.
  35    name : str
  36        Label or dataset identifier to be assigned to all cells in the metadata.
  37
  38    Returns
  39    -------
  40    data : pandas.DataFrame
  41        Dense expression matrix where rows correspond to genes and columns
  42        correspond to cells.
  43    metadata : pandas.DataFrame
  44        Metadata DataFrame with two columns:
  45          - "cell_names": the names of the cells (barcodes)
  46          - "sets": the dataset label assigned to each cell (from `name`)
  47
  48    Notes
  49    -----
  50    The function converts the sparse matrix into a dense DataFrame. This may
  51    require a large amount of memory for datasets with many cells and genes.
  52    """
  53
  54    data = mmread(os.path.join(path, "matrix.mtx"))
  55    data = pd.DataFrame(data.todense())
  56    genes = pd.read_csv(os.path.join(path, "genes.tsv"), header=None, sep="\t")
  57    names = pd.read_csv(os.path.join(path, "barcodes.tsv"), header=None, sep="\t")
  58    data.columns = [str(x) for x in names[0]]
  59    data.index = list(genes[0])
  60
  61    names = list(data.columns)
  62    sets = [name] * len(names)
  63
  64    metadata = pd.DataFrame({"cell_names": names, "sets": sets})
  65
  66    return data, metadata

  81    Generate a volcano plot from differential expression results.

  85    genes that pass significance thresholds.

  96    p_val : float | int, default=0.05
  97        Significance threshold for p-values (or adjusted p-values).
  98    lfc : float | int, default=0.25
  99        Threshold for absolute log fold change.
 100    standard_scale : bool, default=False
 101        If True, use standardized p-values for visualization instead of -log10(p-values).
 102    rescale_adj : bool, default=True
 103        If True, rescale p-values to avoid long breaks caused by outlier values.
 104    image_width : int, default=12
 105        Width of the generated plot in inches.
 106    image_high : int, default=12
 107        Height of the generated plot in inches.
 108
 109    Returns
 110    -------
 111    matplotlib.figure.Figure
 112        The generated volcano plot figure.
 113
 114    """

 440def find_features(data: pd.DataFrame, features: list):
 441    """
 442    Identify features (rows) from a DataFrame that match a given list of features,
 443    ignoring case sensitivity.
 444
 445    Parameters
 446    ----------
 447    data : pandas.DataFrame
 448        DataFrame with features in the index (rows).
 449    features : list
 450        List of feature names to search for.
 451
 452    Returns
 453    -------
 454    dict
 455        Dictionary with keys:
 456        - "included": list of features found in the DataFrame index.
 457        - "not_included": list of requested features not found in the DataFrame index.
 458        - "potential": list of features in the DataFrame that may be similar.
 459    """
 460
 461    features_upper = [str(x).upper() for x in features]
 462
 463    index_set = set(data.index)
 464
 465    features_in = [x for x in index_set if x.upper() in features_upper]
 466    features_in_upper = [x.upper() for x in features_in]
 467    features_out_upper = [x for x in features_upper if x not in features_in_upper]
 468    features_out = [x for x in features if x.upper() in features_out_upper]
 469    similar_features = [
 470        idx for idx in index_set if any(x in idx.upper() for x in features_out_upper)
 471    ]
 472
 473    return {
 474        "included": features_in,
 475        "not_included": features_out,
 476        "potential": similar_features,
 477    }

 516def reduce_data(data: pd.DataFrame, features: list = [], names: list = []):
 517    """
 518    Subset a DataFrame based on selected features (rows) and/or names (columns).
 519
 520    Parameters
 521    ----------
 522    data : pandas.DataFrame
 523        Input DataFrame with features as rows and names as columns.
 524    features : list
 525        List of features to include (rows). Default is an empty list.
 526        If empty, all rows are returned.
 527
 528    names : list
 529        List of names to include (columns). Default is an empty list.
 530        If empty, all columns are returned.
 531
 532    Returns
 533    -------
 534    pandas.DataFrame
 535        Subset of the input DataFrame containing only the selected rows
 536        and/or columns.
 537
 538    Raises
 539    ------
 540    ValueError
 541        If both `features` and `names` are empty.
 542    """
 543
 544    if len(features) > 0 and len(names) > 0:
 545        fet = find_features(data=data, features=features)
 546
 547        nam = find_names(data=data, names=names)
 548
 549        data_to_return = data.loc[fet["included"], nam["included"]]
 550
 551    elif len(features) > 0 and len(names) == 0:
 552        fet = find_features(data=data, features=features)
 553
 554        data_to_return = data.loc[fet["included"], :]
 555
 556    elif len(features) == 0 and len(names) > 0:
 557
 558        nam = find_names(data=data, names=names)
 559
 560        data_to_return = data.loc[:, nam["included"]]
 561
 562    else:
 563
 564        raise ValueError("features and names have zero length!")
 565
 566    return data_to_return

 569def make_unique_list(lst):
 570    """
 571    Generate a list where duplicate items are renamed to ensure uniqueness.
 572
 573    Each duplicate is appended with a suffix ".n", where n indicates the
 574    occurrence count (starting from 1).
 575
 576    Parameters
 577    ----------
 578    lst : list
 579        Input list of items (strings or other hashable types).
 580
 581    Returns
 582    -------
 583    list
 584        List with unique values.
 585
 586    Examples
 587    --------
 588    >>> make_unique_list(["A", "B", "A", "A"])
 589    ['A', 'B', 'A.1', 'A.2']
 590    """
 591    seen = {}
 592    result = []
 593    for item in lst:
 594        if item not in seen:
 595            seen[item] = 0
 596            result.append(item)
 597        else:
 598            seen[item] += 1
 599            result.append(f"{item}.{seen[item]}")
 600    return result

 906def calc_DEG(
 907    data,
 908    metadata_list: list | None = None,
 909    entities: str | list | dict | None = None,
 910    sets: str | list | dict | None = None,
 911    min_exp: int | float = 0,
 912    min_pct: int | float = 0.1,
 913    n_proc: int = 10,
 914):
 915    """
 916    Perform differential gene expression (DEG) analysis on gene expression data.
 917
 918    The function compares groups of cells or samples (defined by `entities` or
 919    `sets`) using the Mann–Whitney U test. It computes p-values, adjusted
 920    p-values, fold changes, standardized effect sizes, and other statistics.
 921
 922    Parameters
 923    ----------
 924    data : pandas.DataFrame
 925        Expression matrix with features (e.g., genes) as rows and samples/cells
 926        as columns.
 927    metadata_list : list or None, optional
 928        Metadata grouping corresponding to the columns in `data`. Required for
 929        comparisons based on sets. Default is None.
 930    entities : list, str, dict, or None, optional
 931        Defines the comparison strategy:
 932        - list of sample names → compare selected cells to the rest.
 933        - 'All' → compare each sample/cell to all others.
 934        - dict → user-defined groups for pairwise comparison.
 935        - None → must be combined with `sets`.
 936    sets : str, dict, or None, optional
 937        Defines group-based comparisons:
 938        - 'All' → compare each set/group to all others.
 939        - dict with two groups → perform pairwise set comparison.
 940        - None → must be combined with `entities`.
 941    min_exp : float | int, default=0
 942        Minimum expression threshold for filtering features.
 943    min_pct : float | int, default=0.1
 944        Minimum proportion of samples within the target group that must express
 945        a feature for it to be tested.
 946    n_proc : int, default=10
 947        Number of parallel processes to use for statistical testing.
 948
 949    Returns
 950    -------
 951    pandas.DataFrame or dict
 952        Results of the differential expression analysis:
 953        - If `entities` is a list → dict with keys: 'valid_cells',
 954          'control_cells', and 'DEG' (results DataFrame).
 955        - If `entities == 'All'` or `sets == 'All'` → DataFrame with results
 956          for all groups.
 957        - If pairwise comparison (dict for `entities` or `sets`) → DataFrame
 958          with results for the specified groups.
 959
 960        The results DataFrame contains:
 961        - 'feature': feature name
 962        - 'p_val': raw p-value
 963        - 'adj_pval': adjusted p-value (multiple testing correction)
 964        - 'pct_valid': fraction of target group expressing the feature
 965        - 'pct_ctrl': fraction of control group expressing the feature
 966        - 'avg_valid': mean expression in target group
 967        - 'avg_ctrl': mean expression in control group
 968        - 'sd_valid': standard deviation in target group
 969        - 'sd_ctrl': standard deviation in control group
 970        - 'esm': effect size metric
 971        - 'FC': fold change
 972        - 'log(FC)': log2-transformed fold change
 973        - 'norm_diff': difference in mean expression
 974
 975    Raises
 976    ------
 977    ValueError
 978        - If `metadata_list` is provided but its length does not match
 979          the number of columns in `data`.
 980        - If neither `entities` nor `sets` is provided.
 981
 982    Notes
 983    -----
 984    - Mann–Whitney U test is used for group comparisons.
 985    - Multiple testing correction is applied using a simple
 986      Benjamini–Hochberg-like method.
 987    - Features expressed below `min_exp` or in fewer than `min_pct` of target
 988      samples are filtered out.
 989    - Parallelization is handled by `joblib.Parallel`.
 990
 991    Examples
 992    --------
 993    Compare a selected list of cells against all others:
 994
 995    >>> result = calc_DEG(data, entities=["cell1", "cell2", "cell3"])
 996
 997    Compare each group to others (based on metadata):
 998
 999    >>> result = calc_DEG(data, metadata_list=group_labels, sets="All")
1000
1001    Perform pairwise comparison between two predefined sets:
1002
1003    >>> sets = {"GroupA": ["A1", "A2"], "GroupB": ["B1", "B2"]}
1004    >>> result = calc_DEG(data, sets=sets)
1005    """
1006
1007    metadata = {}
1008
1009    metadata["primary_names"] = [str(x) for x in data.columns]
1010
1011    if None is not metadata_list:
1012        metadata["sets"] = metadata_list
1013
1014        if len(metadata["primary_names"]) != len(metadata["sets"]):
1015
1016            raise ValueError(
1017                "Metadata list and DataFrame columns must have the same length."
1018            )
1019
1020    else:
1021
1022        metadata["sets"] = [""] * len(metadata["primary_names"])
1023
1024    metadata = pd.DataFrame(metadata)

1026    def stat_calc(choose, feature_name):
1027        target_values = choose.loc[choose["DEG"] == "target", feature_name]
1028        rest_values = choose.loc[choose["DEG"] == "rest", feature_name]
1029
1030        pct_valid = (target_values > 0).sum() / len(target_values)
1031        pct_rest = (rest_values > 0).sum() / len(rest_values)
1032
1033        avg_valid = np.mean(target_values)
1034        avg_ctrl = np.mean(rest_values)
1035        sd_valid = np.std(target_values, ddof=1)
1036        sd_ctrl = np.std(rest_values, ddof=1)
1037        esm = (avg_valid - avg_ctrl) / np.sqrt(((sd_valid**2 + sd_ctrl**2) / 2))
1038
1039        if np.sum(target_values) == np.sum(rest_values):
1040            p_val = 1.0
1041        else:
1042            _, p_val = stats.mannwhitneyu(
1043                target_values, rest_values, alternative="two-sided"
1044            )
1045

1047            "feature": feature_name,
1048            "p_val": p_val,
1049            "pct_valid": pct_valid,
1050            "pct_ctrl": pct_rest,
1051            "avg_valid": avg_valid,
1052            "avg_ctrl": avg_ctrl,
1053            "sd_valid": sd_valid,
1054            "sd_ctrl": sd_ctrl,
1055            "esm": esm,
1056        }

1070
1071        del tmp_dat
1072
1073        drop_col = info["featrue"][info["pct"] <= min_pct]

1083            delayed(stat_calc)(choose, feature)

1085        )
1086
1087        if len(results) > 0:
1088            df = pd.DataFrame(results)
1089            df["valid_group"] = valid_group
1090            df.sort_values(by="p_val", inplace=True)
1091
1092            num_tests = len(df)
1093            df["adj_pval"] = np.minimum(
1094                1, (df["p_val"] * num_tests) / np.arange(1, num_tests + 1)

1096
1097            factors_df = factors.to_frame(name="factor")
1098
1099            df = df.merge(factors_df, left_on="feature", right_index=True, how="left")
1100
1101            df["FC"] = (df["avg_valid"] + df["factor"]) / (
1102                df["avg_ctrl"] + df["factor"]
1103            )
1104            df["log(FC)"] = np.log2(df["FC"])
1105            df["norm_diff"] = df["avg_valid"] - df["avg_ctrl"]
1106            df = df.drop(columns=["factor"])
1107        else:
1108            columns = [
1109                "feature",
1110                "valid_group",
1111                "p_val",
1112                "adj_pval",
1113                "avg_valid",
1114                "avg_ctrl",
1115                "FC",
1116                "log(FC)",
1117                "norm_diff",
1118            ]
1119            df = pd.DataFrame(columns=columns)

1121
